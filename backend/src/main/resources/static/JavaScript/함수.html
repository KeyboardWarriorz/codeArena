자바스크립트에서 함수는 재사용 가능한 코드 블록으로, 특정 작업을 수행하고 값을 반환할 수 있는 독립적인 단위입니다. 함수는 코드의 구조화, 모듈화 및 재사용성을 향상시키는 데 사용됩니다. 아래는 자바스크립트 함수에 대한 주요 개념을 설명합니다.

📒

1. 함수 선언

function functionName(parameter1, parameter2) {
    
        // 함수 본문
    
        }

2. 함수호출

functionName(argument1, argument2);

3. 매개변수와 인수

function greet(name) {
    
        console.log('Hello, ' + name + '!');
    
        }
    
        greet('John'); // "Hello, John!" 출력

4. 반환 값

function add(a, b) {
    
        return a + b;
    
        }
    
        var result = add(2, 3); // result 변수에는 5가 할당됨

5. 함수 표현식

var multiply = function(a, b) {
    
        return a * b;
    
        };
    
        var product = multiply(4, 5); // product 변수에는 20이 할당됨
