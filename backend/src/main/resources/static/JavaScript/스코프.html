스코프(Scope)는 프로그래밍에서 변수의 유효 범위를 나타내는 개념입니다. 스코프는 변수가 선언된 위치와 해당 변수에 접근할 수 있는 범위를 결정합니다. 변수의 스코프에 따라 변수가 어디에서 사용될 수 있는지와 변수의 수명이 결정됩니다.

📒

스코프 사용 예시

// 전역 스코프(Global Scope)
        var globalVariable = 'Global'; // 전역 변수
    
        function outerFunction() {
          // 지역 스코프(Local Scope) - outerFunction
          var outerVariable = 'Outer'; // outerFunction 내부에서만 접근 가능한 변수
    
          function innerFunction() {
            // 지역 스코프(Local Scope) - innerFunction
            var innerVariable = 'Inner'; // innerFunction 내부에서만 접근 가능한 변수
            console.log(globalVariable); // 전역 변수에 접근 가능
            console.log(outerVariable); // 외부 함수의 변수에 접근 가능
            console.log(innerVariable); // 자기 자신의 변수에 접근 가능
          }
    
          innerFunction(); // innerFunction 호출
        }
    
        outerFunction(); // outerFunction 호출
    
        console.log(globalVariable); // 전역 변수에 접근 가능
        console.log(outerVariable); // Error: outerVariable은 outerFunction의 지역 변수로 접근 불가능
        console.log(innerVariable); // Error: innerVariable은 innerFunction의 지역 변수로 접근 불가능

▲ 위의 예시에서는 전역 스코프에 globalVariable이라는 전역 변수를 선언하고, outerFunction이라는 함수 내부에서 outerVariable이라는 지역 변수를 선언합니다. outerFunction 내부에는 innerFunction이라는 함수가 정의되어 있고, innerFunction 내부에서는 innerVariable이라는 지역 변수에 접근합니다.
